```rust
fn wrap_spans(spans: Vec<Span<'static>>, width: usize) -> Vec<Line<'static>> {
    if width == 0 {
        return vec![Line::from(spans)];
    }

    let mut lines = Vec::new();
    let mut current = Vec::new();
    let mut current_width = 0;

    for span in spans {
        let style = span.style;
        for token in split_preserving_spaces(span.content.as_ref()) {
            if token == "\n" {
                push_wrapped_line(&mut lines, &mut current, &mut current_width);
                continue;
            }
            if token.trim().is_empty() {
                if current_width > 0 {
                    current.push(Span::styled(" ".to_string(), style));
                    current_width += 1;
                }
                continue;
            }
            let token_width = UnicodeWidthStr::width(token);
            if current_width > 0 && current_width + token_width > width {
                push_wrapped_line(&mut lines, &mut current, &mut current_width);
            }
            if token_width > width {
                push_split_token(token, style, width, &mut lines, &mut current, &mut current_width);
            } else {
                current.push(Span::styled(token.to_string(), style));
                current_width += token_width;
            }
        }
    }

    push_wrapped_line(&mut lines, &mut current, &mut current_width);
    if lines.is_empty() {
        lines.push(Line::raw(""));
    }
    lines
}

fn split_preserving_spaces(text: &str) -> Vec<&str> {
    let mut tokens = Vec::new();
    let mut start = 0;
    let mut in_space = None;
    for (index, ch) in text.char_indices() {
        if ch == '\n' {
            if start < index {
                tokens.push(&text[start..index]);
            }
            tokens.push("\n");
            start = index + ch.len_utf8();
            in_space = None;
            continue;
        }
        let is_space = ch.is_whitespace();
        if let Some(previous) = in_space {
            if previous != is_space {
                tokens.push(&text[start..index]);
                start = index;
            }
        }
        in_space = Some(is_space);
    }
    if start < text.len() {
        tokens.push(&text[start..]);
    }
    tokens
}

fn push_split_token(
    token: &str,
    style: Style,
    width: usize,
    lines: &mut Vec<Line<'static>>,
    current: &mut Vec<Span<'static>>,
    current_width: &mut usize,
) {
    let mut part = String::new();
    let mut part_width = 0;
    for ch in token.chars() {
        let ch_width = UnicodeWidthChar::width(ch).unwrap_or(0);
        if part_width > 0 && part_width + ch_width > width {
            current.push(Span::styled(std::mem::take(&mut part), style));
            *current_width += part_width;
            push_wrapped_line(lines, current, current_width);
            part_width = 0;
        }
        part.push(ch);
        part_width += ch_width;
    }
    if !part.is_empty() {
        current.push(Span::styled(part, style));
        *current_width += part_width;
    }
}

fn push_wrapped_line(
    lines: &mut Vec<Line<'static>>,
    current: &mut Vec<Span<'static>>,
    current_width: &mut usize,
) {
    if current.is_empty() {
        lines.push(Line::raw(""));
    } else {
        lines.push(Line::from(std::mem::take(current)));
    }
    *current_width = 0;
}
```

```rust
pub(super) fn markdown_lines(text: &str, width: usize) -> Vec<Line<'static>> {
    let text = clean_control_chars(text);
    if text.is_empty() {
        return vec![Line::raw("")];

    let mut renderer = MarkdownRenderer::new(width);

        renderer.push_event(event);
```

```rust
impl MarkdownRenderer {
    fn new(width: usize) -> Self {
        Self {
            width,
            lines: Vec::new(),
            current: Vec::new(),
            styles: vec![TextStyle::default()],
            lists: Vec::new(),
            quote_depth: 0,
            pending_item_prefix: None,
        }
    }

    fn push_event(&mut self, event: Event<'_>) {
        match event {
            Event::Start(Tag::Paragraph) => self.start_text_block(),
            Event::End(TagEnd::Paragraph) => self.end_text_block(),
            Event::Start(Tag::Heading { level, .. }) => self.start_heading(level),
            Event::End(TagEnd::Heading(_)) => self.end_text_block(),
            Event::Start(Tag::BlockQuote(_)) => self.quote_depth += 1,
            Event::End(TagEnd::BlockQuote(_)) => self.quote_depth = self.quote_depth.saturating_sub(1),
            Event::Start(Tag::List(start)) => self.lists.push(ListState { next_number: start }),
            Event::End(TagEnd::List(_)) => {
                self.lists.pop();
                self.ensure_blank_line();
            }
            Event::Start(Tag::Item) => self.start_list_item(),
            Event::End(TagEnd::Item) => self.end_text_block(),
            Event::Start(Tag::Emphasis) => self.push_style(Modifier::ITALIC, None),
            Event::End(TagEnd::Emphasis) => self.pop_style(),
            Event::Start(Tag::Strong) => self.push_style(Modifier::BOLD, None),
            Event::End(TagEnd::Strong) => self.pop_style(),
            Event::Start(Tag::Strikethrough) => self.push_style(Modifier::CROSSED_OUT, None),
            Event::End(TagEnd::Strikethrough) => self.pop_style(),
            Event::Text(text) => self.push_text(text.as_ref()),
            Event::Code(code) => self.push_styled_text(code.as_ref(), inline_code_style()),
            Event::InlineMath(math) => self.push_styled_text(&format!("${math}$"), math_style()),
            Event::DisplayMath(math) => self.push_styled_text(&format!("$${math}$$"), math_style()),
            Event::SoftBreak => self.push_text(" "),
            Event::HardBreak => self.end_text_block(),
            Event::Rule => self.push_rule(),
            Event::TaskListMarker(checked) => self.push_text(if checked { "[x] " } else { "[ ] " }),
            Event::Html(html) | Event::InlineHtml(html) => self.push_styled_text(html.as_ref(), html_style()),
            Event::FootnoteReference(label) => self.push_text(&format!("[^{label}]")),
            Event::Start(_) | Event::End(_) => {}
        }
    }
}
```

Add these helper methods and style functions:

```rust
impl MarkdownRenderer {
    fn start_text_block(&mut self) {
        if self.current.is_empty() {
            for _ in 0..self.quote_depth {
                self.current.push(Span::styled(QUOTE_PREFIX, quote_style()));
            }
            if let Some(prefix) = self.pending_item_prefix.take() {
                self.current.push(Span::raw(prefix));
            }
        }
    }

    fn end_text_block(&mut self) {
        if self.current.is_empty() {
            return;
        }
        let wrapped = wrap_spans(std::mem::take(&mut self.current), self.width);
        self.lines.extend(wrapped);
    }

    fn start_heading(&mut self, level: HeadingLevel) {
        self.start_text_block();
        let marker = "#".repeat(usize::from(level as u8));
        self.current
            .push(Span::styled(format!("{marker} "), heading_style()));
        self.styles.push(TextStyle {
            style: heading_style(),
        });
    }

    fn start_list_item(&mut self) {
        let indent = "  ".repeat(self.lists.len().saturating_sub(1));
        let prefix = if let Some(list) = self.lists.last_mut() {
            if let Some(number) = list.next_number.as_mut() {
                let current = *number;
                *number += 1;
                format!("{indent}{current}. ")
            } else {
                format!("{indent}- ")
            }
        } else {
            "- ".to_string()
        };
        self.pending_item_prefix = Some(prefix);
    }

    fn push_text(&mut self, text: &str) {
        self.start_text_block();
        let style = self.current_style();
        self.current.push(Span::styled(text.to_string(), style));
    }

    fn push_styled_text(&mut self, text: &str, style: Style) {
        self.start_text_block();
        self.current.push(Span::styled(text.to_string(), style));
    }

    fn push_style(&mut self, modifier: Modifier, color: Option<Color>) {
        let mut next = *self.styles.last().unwrap_or(&TextStyle::default());
        next = next.with_modifier(modifier);
        if let Some(color) = color {
            next = next.with_color(color);
        }
        self.styles.push(next);
    }

    fn pop_style(&mut self) {
        if self.styles.len() > 1 {
            self.styles.pop();
        }
    }

    fn current_style(&self) -> Style {
        self.styles.last().copied().unwrap_or_default().style
    }

    fn push_rule(&mut self) {
        self.end_text_block();
        self.lines.push(Line::styled("---", quote_style()));
    }

    fn ensure_blank_line(&mut self) {
        if self.lines.last().is_some_and(|line| !line.spans.is_empty()) {
            self.lines.push(Line::raw(""));
        }
    }

    fn finish(mut self) -> Vec<Line<'static>> {
        self.end_text_block();
        if self.lines.is_empty() {
            self.lines.push(Line::raw(""));
        }
        self.lines
    }
}

fn heading_style() -> Style {
    Style::default()
        .fg(Color::Yellow)
        .add_modifier(Modifier::BOLD)
}

fn quote_style() -> Style {
    Style::default().fg(Color::DarkGray)
}

fn inline_code_style() -> Style {
    Style::default().fg(Color::Magenta)
}

fn math_style() -> Style {
    Style::default().fg(Color::Cyan)
}

fn html_style() -> Style {
    Style::default().fg(Color::DarkGray)
}
```

        self.end_text_block();
        let language = match kind {
            CodeBlockKind::Fenced(info) => info.split_whitespace().next().unwrap_or("").to_string(),
            CodeBlockKind::Indented => String::new(),
        };
        self.code_block = Some(CodeBlock {
            language,
            text: String::new(),
        });
    }

    fn push_code_text(&mut self, text: &str) {
        if let Some(block) = &mut self.code_block {
            block.text.push_str(text);
        }
    }

    fn end_code_block(&mut self) {
        let Some(block) = self.code_block.take() else {
            return;
        };
        let fence = if block.language.is_empty() {
            "```".to_string()
        } else {
            format!("```{}", block.language)
        };
        self.lines.push(Line::styled(fence, code_style()));
        for raw_line in block.text.lines() {
            self.lines.extend(wrap_spans(
                vec![Span::styled(raw_line.to_string(), code_style())],
                self.width,
            ));
        }
        self.lines.push(Line::styled("```", code_style()));
    }
}

fn code_style() -> Style {
    Style::default().fg(Color::Gray)
}

fn link_style() -> Style {
    Style::default()
        .fg(Color::Blue)
        .add_modifier(Modifier::UNDERLINED)
}

```rust
impl MarkdownRenderer {
    fn start_table_row(&mut self) {
        if let Some(table) = &mut self.table {
            table.current_row.clear();
        }
    }

    fn end_table_row(&mut self) {
        if let Some(table) = &mut self.table {
            table.rows.push(std::mem::take(&mut table.current_row));
        }
    }

    fn start_table_cell(&mut self) {
        if let Some(table) = &mut self.table {
            table.in_cell = true;
            table.current_cell.clear();
        }
    }

    fn end_table_cell(&mut self) {
        if let Some(table) = &mut self.table {
            table.in_cell = false;
            table
                .current_row
                .push(table.current_cell.trim().to_string());
        }
    }

    fn push_table_text(&mut self, text: &str) {
        if let Some(table) = &mut self.table {
            table.current_cell.push_str(text);
        }
    }

    fn flush_table(&mut self) {
        let Some(table) = self.table.take() else {
            return;
        };
        let column_count = table.rows.iter().map(Vec::len).max().unwrap_or(0);
        let mut widths = vec![3; column_count];
        for row in &table.rows {
            for (index, cell) in row.iter().enumerate() {
                widths[index] = widths[index].max(UnicodeWidthStr::width(cell.as_str()));
            }
        }
        for (row_index, row) in table.rows.iter().enumerate() {
            self.lines.push(Line::raw(render_table_row(row, &widths)));
            if row_index == 0 {
                let separator = widths
                    .iter()
                    .map(|width| "-".repeat(*width))
                    .collect::<Vec<_>>();
                self.lines.push(Line::raw(render_table_row(&separator, &widths)));
            }
        }
    }
}

fn render_table_row(row: &[String], widths: &[usize]) -> String {
    let mut output = String::from("|");
    for (index, width) in widths.iter().enumerate() {
        let cell = row.get(index).map_or("", String::as_str);
        let padding = width.saturating_sub(UnicodeWidthStr::width(cell));
        output.push(' ');
        output.push_str(cell);
        output.push_str(&" ".repeat(padding));
        output.push(' ');
        output.push('|');
    }
    output
}
```
